    private static int getKey(Map<Integer, String> obj){
        Set<Integer> keySet = obj.keySet();
        int firstKey = -1;
        for(int key : keySet){
            firstKey = key;
            if(firstKey != -1){
                break;
            }
        }
        return firstKey;
    }
